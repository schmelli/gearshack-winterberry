### conversation_memory Table

```sql
CREATE TABLE conversation_memory (
  -- Primary key
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  -- Foreign keys
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,

  -- Conversation grouping
  conversation_id UUID NOT NULL,

  -- Message identity (globally unique across all conversations)
  message_id UUID NOT NULL UNIQUE,

  -- Message content
  role TEXT NOT NULL CHECK (role IN ('user', 'assistant', 'system')),
  content TEXT NOT NULL,

  -- Flexible metadata storage (tool calls, citations, preferences)
  metadata JSONB DEFAULT '{}'::jsonb,

  -- Timestamps for conflict resolution and retention
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),

  -- Composite unique constraint for conflict detection
  CONSTRAINT unique_user_conversation_message UNIQUE (user_id, conversation_id, message_id)
);

-- Performance indexes
CREATE INDEX idx_conversation_memory_lookup
  ON conversation_memory (user_id, conversation_id, created_at DESC);

CREATE INDEX idx_conversation_memory_metadata
  ON conversation_memory USING gin(metadata jsonb_path_ops);

-- Full-text search index (for semantic search Phase 2+)
CREATE INDEX idx_conversation_memory_content_search
  ON conversation_memory USING gin(to_tsvector('english', content));

-- Row Level Security
ALTER TABLE conversation_memory ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can only access own memory"
  ON conversation_memory
  FOR ALL
  USING (auth.uid() = user_id);

  ON conversation_memory
  FOR ALL
  TO service_role
  USING (true);
```

## Supabase Memory Adapter Implementation

### File Location
`lib/mastra/adapters/supabase-memory-adapter.ts`

### Complete Implementation

```typescript
import { SupabaseClient } from '@supabase/supabase-js';
import { MemoryAdapter, Message } from '@mastra/memory';

export class SupabaseMemoryAdapter implements MemoryAdapter {
  private supabase: SupabaseClient;

  constructor(supabaseClient: SupabaseClient) {
    this.supabase = supabaseClient;
  }

  /**
   * Save messages with last-write-wins conflict resolution
   */
  async saveMessages(messages: Message[]): Promise<void> {
    const records = messages.map(msg => ({
      user_id: msg.userId,
      conversation_id: msg.conversationId,
      message_id: msg.id,
      role: msg.role,
      content: msg.content,
      metadata: msg.metadata || {},
      created_at: msg.createdAt.toISOString(),
      updated_at: new Date().toISOString(), // Server-side timestamp for conflict resolution
    }));

    const { error } = await this.supabase
      .from('conversation_memory')
      .upsert(records, {
        onConflict: 'message_id', // Unique constraint
        ignoreDuplicates: false, // Always overwrite with latest
      });

    if (error) {
      throw new Error(`Failed to save messages: ${error.message}`);
    }
  }

  /**
   * Retrieve conversation history with optional filtering
   */
  async getMessages(options: {
    userId: string;
    conversationId?: string;
    limit?: number;
    before?: Date;
  }): Promise<Message[]> {
    let query = this.supabase
      .from('conversation_memory')
      .select('*')
      .eq('user_id', options.userId)
      .order('created_at', { ascending: false });

    // Filter by conversation if specified
    if (options.conversationId) {
      query = query.eq('conversation_id', options.conversationId);
    }

    // Filter messages before timestamp (for pagination)
    if (options.before) {
      query = query.lt('created_at', options.before.toISOString());
    }

    // Limit results (default 100, max 1000)
    const limit = Math.min(options.limit || 100, 1000);
    query = query.limit(limit);

    const { data, error } = await query;

    if (error) {
      throw new Error(`Failed to retrieve messages: ${error.message}`);
    }

    // Map database records to Message interface
    return (data || []).map(record => ({
      id: record.message_id,
      userId: record.user_id,
      conversationId: record.conversation_id,
      role: record.role as 'user' | 'assistant' | 'system',
      content: record.content,
      metadata: record.metadata,
      createdAt: new Date(record.created_at),
      updatedAt: new Date(record.updated_at),
    }));
  }

  /**
   * Delete conversation or specific messages (GDPR compliance)
   */
  async deleteMessages(options: {
    userId: string;
    conversationId?: string;
    messageIds?: string[];
  }): Promise<void> {
    let query = this.supabase
      .from('conversation_memory')
      .delete()
      .eq('user_id', options.userId);

    // Delete specific conversation
    if (options.conversationId && !options.messageIds) {
      query = query.eq('conversation_id', options.conversationId);
    }

    // Delete specific messages
    if (options.messageIds && options.messageIds.length > 0) {
      query = query.in('message_id', options.messageIds);
    }

    // If neither conversationId nor messageIds provided, delete ALL user memory
    // (GDPR Right to Erasure)

    const { error } = await query;

    if (error) {
      throw new Error(`Failed to delete messages: ${error.message}`);
    }
  }

  /**
   * Semantic search across conversations (Phase 2+ with pgvector)
   * MVP: Use full-text search as fallback
   */
  async searchMessages(
    userId: string,
    query: string,
    limit: number = 10
  ): Promise<Message[]> {
    // Phase 1: Full-text search using PostgreSQL tsvector
    const { data, error } = await this.supabase
      .from('conversation_memory')
      .select('*')
      .eq('user_id', userId)
      .textSearch('content', query, { type: 'websearch' })
      .order('created_at', { ascending: false })
      .limit(limit);

    if (error) {
      throw new Error(`Failed to search messages: ${error.message}`);
    }

    return (data || []).map(record => ({
      id: record.message_id,
      userId: record.user_id,
      conversationId: record.conversation_id,
      role: record.role as 'user' | 'assistant' | 'system',
      content: record.content,
      metadata: record.metadata,
      createdAt: new Date(record.created_at),
      updatedAt: new Date(record.updated_at),
    }));

    // Phase 2+: Replace with pgvector semantic search
    // const { data } = await this.supabase.rpc('search_conversation_memory', {
    //   query_embedding: await generateEmbedding(query),
    //   match_threshold: 0.7,
    //   match_count: limit,
    //   user_id: userId
    // });
  }
}
```

## Conflict Resolution Strategy

### Last-Write-Wins (LWW) Implementation

**Scenario**: User updates the same message from multiple devices simultaneously.

#### Example Conflict
1. **Device A** (phone) sends message update at `T1` (server receives at `T1+100ms`)
2. **Device B** (tablet) sends same message update at `T2` (server receives at `T2+50ms`, arrives first)
3. Both devices have different `content` for the same `message_id`

#### Resolution Mechanism
```sql
-- Supabase upsert with ON CONFLICT behavior
INSERT INTO conversation_memory (
  user_id, conversation_id, message_id, role, content, metadata, created_at, updated_at
) VALUES (
  $1, $2, $3, $4, $5, $6, $7, now() -- Server timestamp
)
ON CONFLICT (message_id) DO UPDATE SET
  content = EXCLUDED.content,
  metadata = EXCLUDED.metadata,
  updated_at = now(); -- Always use server timestamp
```

**Outcome**: Final state = **Device A's update** (later `updated_at` timestamp)

#### Why Server Timestamps?
- Client clocks may be out of sync (NTP drift, timezone issues)
- Server-side `now()` provides authoritative ordering
- PostgreSQL transaction isolation ensures atomic updates

```typescript
// lib/mastra/retention.ts
export async function cleanupExpiredMemory(supabase: SupabaseClient) {
  const retentionDays = 90;
  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - retentionDays);

  const { error } = await supabase
    .from('conversation_memory')
    .delete()
    .lt('created_at', cutoffDate.toISOString());

  if (error) {
    throw new Error(`Retention cleanup failed: ${error.message}`);

}
```

```typescript
// __tests__/lib/mastra/supabase-memory-adapter.test.ts
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { SupabaseMemoryAdapter } from '@/lib/mastra/adapters/supabase-memory-adapter';
import { createClient } from '@supabase/supabase-js';

describe('SupabaseMemoryAdapter', () => {
  let adapter: SupabaseMemoryAdapter;
  let testUserId: string;

  beforeEach(async () => {
    const supabase = createClient(/* test credentials */);
    adapter = new SupabaseMemoryAdapter(supabase);
    testUserId = 'test-user-' + Date.now();
  });

  afterEach(async () => {
    // Cleanup test data
    await adapter.deleteMessages({ userId: testUserId });
  });

  it('should save and retrieve messages', async () => {
    const messages = [
      {
        id: 'msg-1',
        userId: testUserId,
        conversationId: 'conv-1',
        role: 'user' as const,
        content: 'Hello',
        createdAt: new Date(),
        updatedAt: new Date(),
      }
    ];

    await adapter.saveMessages(messages);
    const retrieved = await adapter.getMessages({ userId: testUserId, conversationId: 'conv-1' });

    expect(retrieved).toHaveLength(1);
    expect(retrieved[0].content).toBe('Hello');
  });

  it('should handle concurrent updates with last-write-wins', async () => {
    const messageId = 'msg-conflict-test';

    // Simulate Device A update
    await adapter.saveMessages([{
      id: messageId,
      userId: testUserId,
      conversationId: 'conv-1',
      role: 'user',
      content: 'Device A content',
      createdAt: new Date(),
      updatedAt: new Date(),
    }]);

    // Wait 100ms, then Device B update
    await new Promise(resolve => setTimeout(resolve, 100));

    await adapter.saveMessages([{
      id: messageId,
      userId: testUserId,
      conversationId: 'conv-1',
      role: 'user',
      content: 'Device B content (later)',
      createdAt: new Date(),
      updatedAt: new Date(),
    }]);

    const result = await adapter.getMessages({ userId: testUserId });
    expect(result[0].content).toBe('Device B content (later)'); // Last write wins
  });

  it('should delete conversation completely (GDPR compliance)', async () => {
    await adapter.saveMessages([
      { id: 'msg-1', userId: testUserId, conversationId: 'conv-1', role: 'user', content: 'Test', createdAt: new Date(), updatedAt: new Date() },
      { id: 'msg-2', userId: testUserId, conversationId: 'conv-1', role: 'assistant', content: 'Response', createdAt: new Date(), updatedAt: new Date() }
    ]);

    await adapter.deleteMessages({ userId: testUserId, conversationId: 'conv-1' });

    const remaining = await adapter.getMessages({ userId: testUserId, conversationId: 'conv-1' });
    expect(remaining).toHaveLength(0);
  });

  it('should search messages by full-text query', async () => {
    await adapter.saveMessages([
      { id: 'msg-1', userId: testUserId, conversationId: 'conv-1', role: 'user', content: 'What is the lightest tent?', createdAt: new Date(), updatedAt: new Date() },
      { id: 'msg-2', userId: testUserId, conversationId: 'conv-1', role: 'assistant', content: 'The Nemo Hornet Elite weighs 850g', createdAt: new Date(), updatedAt: new Date() }
    ]);

    const results = await adapter.searchMessages(testUserId, 'lightest tent');
    expect(results.length).toBeGreaterThan(0);
    expect(results[0].content).toContain('tent');
  });
});
```

## Phase 2+ Enhancements

### Semantic Search with pgvector

```sql
-- Add vector embedding column (Phase 2)
ALTER TABLE conversation_memory ADD COLUMN embedding vector(1536); -- OpenAI ada-002 dimension

-- Create vector similarity index
CREATE INDEX idx_conversation_memory_embedding ON conversation_memory USING ivfflat (embedding vector_cosine_ops);

-- Semantic search function
CREATE OR REPLACE FUNCTION search_conversation_memory(
  query_embedding vector(1536),
  match_threshold float,
  match_count int,
  user_id uuid
)
RETURNS TABLE (
  message_id uuid,
  content text,
  similarity float
)
LANGUAGE plpgsql
AS $$
BEGIN
  RETURN QUERY
  SELECT
    cm.message_id,
    cm.content,
    1 - (cm.embedding <=> query_embedding) AS similarity
  FROM conversation_memory cm
  WHERE cm.user_id = user_id
    AND 1 - (cm.embedding <=> query_embedding) > match_threshold
  ORDER BY cm.embedding <=> query_embedding
  LIMIT match_count;
END;
$$;
```
